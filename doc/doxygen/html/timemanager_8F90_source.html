241  end do

292  !**************************************************
293

516  endif

529 
530  if (ipout.eq.2) call partoutput(itime) ! dump particle positions

532  if (linit_cond.ge.1) call initial_cond_output(itime) ! dump initial cond. field

559  deallocate(oroout, area, volume)
